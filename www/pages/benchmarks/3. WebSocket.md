```sql query
SELECT "server" as server, 
    "Number of clients", 
    "Requests per second (mean)",
    "Requests per second (max)",
    "Number of 2xx responses"
FROM results.data
WHERE "Test endopint" = '${inputs.test_endpoint}'
AND "Protocol" = '${protocol}'
AND "Treeish" in ('master', 'main')
```
